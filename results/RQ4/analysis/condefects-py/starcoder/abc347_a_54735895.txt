No.0 Patch
n, k = map(int, input().split())
a = list(map(int, input().split()))
b = []

for i in range(n):
  if a[i] % k == 0:
      b.append(a[i] // k)

b.sort()
print(*b)

# Analysis

# n = Length of array
# k = dividing factor

# Time Complexity: O(nlogn)
# Space Complexity: O(n)
